
E -> C | C + C | E + C | E + E

C -> K + V | K + DC | K + F | K + FVP | KC + V | KC + DC | KC + F | KC + FVP

KC -> n * K

DC -> n * FVP

FVP -> F + V

########################################
# Non terminals
########################################

E = Expression

C = Command

KC = KeywordChain

FVP = FlagValuePair

DC = DataChain

########################################
#terminals
########################################

K = Keyword

V = Value

F = Flag
