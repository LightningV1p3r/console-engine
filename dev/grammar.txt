
E -> C | C + C | E + C | E + E

C -> K + V | K + DC | K + F | K + FVP | KC + V | KC + DC | KC + F | KC + FVP

KC -> n * K

DC -> n * FVP | FC + n * FVP

FC -> n * F

FVP -> F + V

########################################
# Non terminals
########################################

E = Expression

C = Command

FC = FlagChain

KC = KeywordChain

FVP = FlagValuePair

DC = DataChain

########################################
#terminals
########################################

K = Keyword

V = Value

F = Flag
